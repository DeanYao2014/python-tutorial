# 6.12 图算法与字符串算法

本章是算法之旅的收尾——Dijkstra 最短路径、拓扑排序、KMP 字符串匹配、Trie 前缀树、并查集。每个算法都是实际工程中的高频工具。

## 1.1 Dijkstra 最短路径

**Dijkstra = BFS + 优先队列**。用堆确保每次取出距离起点最近的节点。

```python-run
import heapq
from collections import defaultdict

def dijkstra(graph, start):
    """graph: {node: [(neighbor, weight), ...]}"""
    dist = {start: 0}
    heap = [(0, start)]                    # (距离, 节点)
    prev = {}                              # 记录路径来源

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist.get(u, float('inf')):  # 跳过过时的记录
            continue

        for v, w in graph[u]:
            new_dist = d + w
            if new_dist < dist.get(v, float('inf')):
                dist[v] = new_dist
                prev[v] = u
                heapq.heappush(heap, (new_dist, v))
    return dist, prev

# 构建加权图
g = {
    'A': [('B', 4), ('C', 2)],
    'B': [('C', 1), ('D', 5)],
    'C': [('D', 8), ('E', 10)],
    'D': [('E', 2)],
    'E': [],
}

dist, prev = dijkstra(g, 'A')
print(f"从 A 出发的最短距离: {dist}")

# 回溯构建最短路径 A → E
def get_path(prev, target):
    path = []
    cur = target
    while cur in prev:
        path.append(cur)
        cur = prev[cur]
    path.append(cur)
    return path[::-1]

print(f"A→E 最短路径: {' → '.join(get_path(prev, 'E'))}")
```

**复杂度**：O((V + E) log V)。每条边最多入堆一次，堆操作 O(log V)。

> Dijkstra 要求**边权非负**。有负权边用 **Bellman-Ford**，全源最短路径用 **Floyd-Warshall**。

## 1.2 拓扑排序

DAG（有向无环图）才有拓扑序。两种方法：Kahn 算法（BFS 统计入度）和 DFS 后序遍历。

```python-run
from collections import deque, defaultdict

def topological_sort(graph):
    """Kahn 算法：BFS + 入度表。graph 是有向图"""
    # 计算入度
    indegree = {node: 0 for node in graph}
    for u in graph:
        for v in graph[u]:
            indegree[v] = indegree.get(v, 0) + 1

    # 入度为 0 的节点入队
    q = deque([node for node in indegree if indegree[node] == 0])
    result = []

    while q:
        u = q.popleft()
        result.append(u)
        for v in graph[u]:
            indegree[v] -= 1
            if indegree[v] == 0:
                q.append(v)

    if len(result) != len(indegree):
        return None              # 有环 → 无法拓扑排序
    return result

# 依赖关系：A→B, A→C, B→D, C→D, D→E
dag = {
    'A': ['B', 'C'],
    'B': ['D'],
    'C': ['D'],
    'D': ['E'],
    'E': [],
}
print(f"拓扑序: {' → '.join(topological_sort(dag))}")
```

## 1.3 KMP 字符串匹配

朴素匹配：每轮失配后主串回退一位，子串从头开始——O(n × m)。**KMP 利用失配字符的前后缀信息跳过无意义的比较，O(n + m)**。

```python-run
def build_lps(pattern):
    """构建最长公共前后缀表（LPS）"""
    lps = [0] * len(pattern)
    length = 0                     # 当前最长前后缀长度
    i = 1
    while i < len(pattern):
        if pattern[i] == pattern[length]:
            length += 1
            lps[i] = length
            i += 1
        elif length > 0:
            length = lps[length - 1]   # 回退
        else:
            lps[i] = 0
            i += 1
    return lps

def kmp(text, pattern):
    """KMP 字符串匹配，返回所有匹配位置"""
    if not pattern:
        return [0]
    lps = build_lps(pattern)
    result = []
    i = j = 0                      # i → text, j → pattern
    while i < len(text):
        if text[i] == pattern[j]:
            i += 1; j += 1
            if j == len(pattern):
                result.append(i - j)
                j = lps[j - 1]     # 继续找下一个
        elif j > 0:
            j = lps[j - 1]         # 关键：跳过失配已匹配的前缀
        else:
            i += 1
    return result

text = "ABABDABACDABABCABAB"
pattern = "ABABCABAB"
print(f"匹配位置: {kmp(text, pattern)}")

# KMP 的核心：LPS 告诉我们"失配时，pattern 的哪个前缀已经是 text 的一部分"
```

> Python 的 `str.find()` 内部使用 Fast Search（Boyer-Moore 简化版），比 KMP 在实际中更快。KMP 的价值在于其**确定性 O(n)** 和理论基础。

## 1.4 Trie（前缀树）

Trie 按字符前缀组织字符串集合——支持高效的**前缀匹配、自动补全、词频统计**。

```python-run
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False         # 该节点是否是某个词的结束

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word: str) -> bool:
        """精确查找"""
        node = self.root
        for ch in word:
            if ch not in node.children:
                return False
            node = node.children[ch]
        return node.is_end

    def starts_with(self, prefix: str) -> bool:
        """是否存在以 prefix 开头的词"""
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return False
            node = node.children[ch]
        return True

# 演示
trie = Trie()
for w in ["cat", "car", "care", "dog", "door"]:
    trie.insert(w)

print(f"search('cat'): {trie.search('cat')}")
print(f"search('ca'): {trie.search('ca')}")       # False — 'ca' 没有被插入
print(f"starts_with('ca'): {trie.starts_with('ca')}")   # True — 'cat', 'car', 'care'
print(f"starts_with('do'): {trie.starts_with('do')}")   # True — 'dog', 'door'
```

## 1.5 并查集（Union-Find）

并查集解决"判断两个元素是否在同一组"的问题——近乎 O(1) 的合并与查询。

```python-run
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n          # 按秩合并

    def find(self, x):
        """路径压缩：把路径上所有节点直接指向根"""
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):
        """按秩合并：小的挂在大的下面"""
        rx, ry = self.find(x), self.find(y)
        if rx == ry:
            return False             # 已在同一组
        if self.rank[rx] < self.rank[ry]:
            self.parent[rx] = ry
        elif self.rank[rx] > self.rank[ry]:
            self.parent[ry] = rx
        else:
            self.parent[ry] = rx
            self.rank[rx] += 1
        return True

    def connected(self, x, y):
        return self.find(x) == self.find(y)

# 演示：朋友圈问题
uf = UnionFind(5)
uf.union(0, 1)   # 0 和 1 是朋友
uf.union(1, 2)   # 1 和 2 是朋友 → 0,1,2 在同一组
uf.union(3, 4)   # 3 和 4 是朋友

print(f"0,2 是否同组: {uf.connected(0, 2)}")   # True
print(f"0,3 是否同组: {uf.connected(0, 3)}")   # False
print(f"组数: {len(set(uf.find(i) for i in range(5)))}")  # 2
```

## 本章小结

1. **Dijkstra = BFS + 最小堆**，O((V+E) log V)，要求边权非负
2. **拓扑排序 = BFS + 入度表**（Kahn 算法），只适用于 DAG；入度为 0 的节点依次输出
3. **KMP 的核心是 LPS 表**——失配时用已匹配部分的前后缀信息跳跃，O(n+m)
4. **Trie 适合前缀匹配场景**：搜索提示、拼写检查、IP 路由表
5. **并查集是连通性问题的终极工具**：路径压缩 + 按秩合并 → 均摊近乎 O(1)
